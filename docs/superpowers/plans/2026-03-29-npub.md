# npub Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone Go binary that builds a static site from a local notes store and serves it locally for preview, replacing the existing Ruby-based builder.

**Architecture:** The binary has two commands (`build`, `serve`). `build` scans a notes store via `notescli/note`, parses frontmatter, renders Markdown with Goldmark (including AST transformers for images and note links), and writes HTML pages to a dist directory using `html/template`. `serve` is a thin `net/http` file server. Configuration uses three-layer precedence: flags > env vars > YAML file.

**Tech Stack:** Go 1.25+, `github.com/dreikanter/notescli/note` v0.1.36, `github.com/spf13/cobra`, `github.com/yuin/goldmark` + `github.com/alecthomas/chroma`, `gopkg.in/yaml.v3`, Tailwind CSS 3.x for stylesheets.

```go
package config

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"

	"gopkg.in/yaml.v3"
)

// Config holds all configuration for a npub build.
type Config struct {
	NotesPath   string `yaml:"notes_path"`
	AssetsPath  string `yaml:"assets_path"`
	BuildPath   string `yaml:"build_path"`
	SiteRootURL string `yaml:"site_root_url"`
	SiteName    string `yaml:"site_name"`
	AuthorName  string `yaml:"author_name"`
}

// SiteRootPath returns the URL path component of SiteRootURL.
// For "https://example.com" it returns "/".
// For "https://example.com/blog" it returns "/blog".
func (c Config) SiteRootPath() string {
	idx := strings.Index(c.SiteRootURL, "://")
	if idx < 0 {
		return "/"
	}
	rest := c.SiteRootURL[idx+3:]
	slash := strings.IndexByte(rest, '/')
	if slash < 0 {
		return "/"
	}
	p := rest[slash:]
	p = strings.TrimRight(p, "/")
	if p == "" {
		return "/"
	}
	return p
}

// FeedURL returns the full feed URL.
func (c Config) FeedURL() string {
	return strings.TrimRight(c.SiteRootURL, "/") + c.FeedPath()
}

// FeedPath returns the path to the feed relative to site root.
func (c Config) FeedPath() string {
	root := c.SiteRootPath()
	if root == "/" {
		return "/feed.xml"
	}
	return root + "/feed.xml"
}

// Load reads configuration from a YAML file, then overlays environment variables,
// then overlays flag values. envOverrides maps env var names to values.
// flagOverrides maps flag names to values.
func Load(yamlPath string, envOverrides map[string]string, flagOverrides map[string]string) (Config, error) {
	data, err := os.ReadFile(yamlPath)
	if err != nil {
		return Config{}, fmt.Errorf("cannot read config: %w", err)
	}

	var cfg Config
	if err := yaml.Unmarshal(data, &cfg); err != nil {
		return Config{}, fmt.Errorf("cannot parse config: %w", err)
	}

	// Env overrides.
	envMap := map[string]*string{
		"NOTES_PATH":             &cfg.NotesPath,
		"NPUB_ASSETS_PATH":   &cfg.AssetsPath,
		"NPUB_BUILD_PATH":    &cfg.BuildPath,
		"NPUB_SITE_ROOT_URL": &cfg.SiteRootURL,
		"NPUB_SITE_NAME":     &cfg.SiteName,
		"NPUB_AUTHOR_NAME":   &cfg.AuthorName,
	}
	for envKey, ptr := range envMap {
		if v, ok := envOverrides[envKey]; ok && v != "" {
			*ptr = v
		}
	}

	// Flag overrides.
	flagMap := map[string]*string{
		"notes-path": &cfg.NotesPath,
		"assets-path": &cfg.AssetsPath,
		"out":         &cfg.BuildPath,
		"url":         &cfg.SiteRootURL,
		"site-name":   &cfg.SiteName,
		"author":      &cfg.AuthorName,
	}
	for flagKey, ptr := range flagMap {
		if v, ok := flagOverrides[flagKey]; ok && v != "" {
			*ptr = v
		}
	}

	// Expand ~ in paths.
	cfg.NotesPath = expandHome(cfg.NotesPath)
	cfg.AssetsPath = expandHome(cfg.AssetsPath)
	cfg.BuildPath = expandHome(cfg.BuildPath)

	// Validate required fields.
	if cfg.SiteRootURL == "" {
		return Config{}, fmt.Errorf("site_root_url is required")
	}
	if cfg.SiteName == "" {
		return Config{}, fmt.Errorf("site_name is required")
	}
	if cfg.AuthorName == "" {
		return Config{}, fmt.Errorf("author_name is required")
	}

	return cfg, nil
}

func expandHome(path string) string {
	if !strings.HasPrefix(path, "~/") {
		return path
	}
	home, err := os.UserHomeDir()
	if err != nil {
		return path
	}
	return filepath.Join(home, path[2:])
}
```

```go
package page

import (
	"fmt"
	"path"
	"sort"
	"strings"
	"time"
)

// NotePage represents a single public note.
type NotePage struct {
	UID         string
	ShortUID    string
	Slug        string
	Title       string
	Description string
	Tags        []string
	Body        string // rendered HTML
	PublishedAt time.Time
	SiteRootURL string
	Attachments []Attachment
}

// Attachment is a downloaded image file associated with a note.
type Attachment struct {
	FileName string
	PageUID  string
}

// LocalPath returns the output file path relative to build_path.
func (p NotePage) LocalPath() string {
	return path.Join(p.UID, p.Slug, "index.html")
}

// PublicPath returns the URL path for this note.
func (p NotePage) PublicPath() string {
	return path.Join(p.UID, p.Slug)
}

// URL returns the full URL for this note.
func (p NotePage) URL() string {
	return strings.TrimRight(p.SiteRootURL, "/") + "/" + p.PublicPath()
}

// CanonicalPath returns the canonical URL path.
func (p NotePage) CanonicalPath() string {
	return p.PublicPath()
}

// SortNotePages sorts notes by PublishedAt descending (newest first).
func SortNotePages(pages []NotePage) {
	sort.Slice(pages, func(i, j int) bool {
		return pages[i].PublishedAt.After(pages[j].PublishedAt)
	})
}

// RelatedTo returns notes that share at least one tag with the given page.
func RelatedTo(pages []NotePage, target NotePage) []NotePage {
	tagSet := make(map[string]struct{}, len(target.Tags))
	for _, t := range target.Tags {
		tagSet[t] = struct{}{}
	}
	var related []NotePage
	for _, p := range pages {
		if p.UID == target.UID {
			continue
		}
		for _, t := range p.Tags {
			if _, ok := tagSet[t]; ok {
				related = append(related, p)
				break
			}
		}
	}
	return related
}

// TaggedPages returns all notes that have the given tag.
func TaggedPages(pages []NotePage, tag string) []NotePage {
	var result []NotePage
	for _, p := range pages {
		for _, t := range p.Tags {
			if t == tag {
				result = append(result, p)
				break
			}
		}
	}
	return result
}

// AllTags returns a sorted, deduplicated list of all tags across all notes.
func AllTags(pages []NotePage) []string {
	seen := make(map[string]struct{})
	for _, p := range pages {
		for _, t := range p.Tags {
			seen[t] = struct{}{}
		}
	}
	tags := make([]string, 0, len(seen))
	for t := range seen {
		tags = append(tags, t)
	}
	sort.Strings(tags)
	return tags
}

// RedirectPage is an HTML redirect from UID path to the full slug path.
type RedirectPage struct {
	UID        string
	RedirectTo string
}

// LocalPath returns the output file path.
func (p RedirectPage) LocalPath() string {
	return path.Join(p.UID, "index.html")
}

// TagPage lists notes filtered by a single tag.
type TagPage struct {
	Tag string
}

// LocalPath returns the output file path.
func (p TagPage) LocalPath() string {
	return fmt.Sprintf("tags/%s/index.html", p.Tag)
}

// PublicPath returns the URL path for this tag page.
func (p TagPage) PublicPath() string {
	return fmt.Sprintf("tags/%s", p.Tag)
}

// CanonicalPath returns the canonical URL path.
func (p TagPage) CanonicalPath() string {
	return p.PublicPath()
}
```

```go
package render

import (
	"bytes"
	"regexp"

	"github.com/yuin/goldmark"
	"github.com/yuin/goldmark/ast"
	"github.com/yuin/goldmark/extension"
	"github.com/yuin/goldmark/parser"
	"github.com/yuin/goldmark/text"
	"github.com/yuin/goldmark/util"

	highlighting "github.com/yuin/goldmark-highlighting/v2"
)

// ProcessImageFunc is called for each external image URL during rendering.
// It returns the local filename to use instead.
type ProcessImageFunc func(src string) (localName string, err error)

// Render converts Markdown to HTML.
// noteIndex maps note IDs to their public paths (for link resolution).
// processImage is called for external image URLs (may be nil to skip image processing).
func Render(source []byte, noteIndex map[string]string, processImage ProcessImageFunc) ([]byte, error) {
	md := goldmark.New(
		goldmark.WithExtensions(
			extension.GFM,
			highlighting.NewHighlighting(
				highlighting.WithStyle("github"),
				highlighting.WithFormatOptions(),
			),
		),
		goldmark.WithParserOptions(
			parser.WithAutoHeadingID(),
		),
	)

	reader := text.NewReader(source)
	doc := md.Parser().Parse(reader)

	// Walk AST and transform links and images.
	err := ast.Walk(doc, func(n ast.Node, entering bool) (ast.WalkStatus, error) {
		if !entering {
			return ast.WalkContinue, nil
		}

		switch node := n.(type) {
		case *ast.Link:
			dest := string(node.Destination)
			if isNoteID(dest) && noteIndex != nil {
				if resolved, ok := noteIndex[dest]; ok {
					node.Destination = []byte("/" + resolved)
				}
			}

		case *ast.Image:
			src := string(node.Destination)
			if processImage != nil && isExternalURL(src) {
				localName, err := processImage(src)
				if err != nil {
					return ast.WalkContinue, nil // leave URL as-is on error
				}
				node.Destination = []byte(localName)
			}
		}

		return ast.WalkContinue, nil
	})
	if err != nil {
		return nil, err
	}

	var buf bytes.Buffer
	if err := md.Renderer().Render(&buf, source, doc); err != nil {
		return nil, err
	}
	return buf.Bytes(), nil
}

var noteIDPattern = regexp.MustCompile(`^\d+$`)

func isNoteID(s string) bool {
	return noteIDPattern.MatchString(s)
}

func isExternalURL(s string) bool {
	return len(s) > 8 && (s[:8] == "https://" || s[:7] == "http://")
}

// Ensure the AST transformer interface is satisfied at compile time.
var _ ast.Walker = (ast.Walker)(nil)
// Note: We walk the AST manually rather than implementing a transformer,
// because goldmark's transformer API registers at parse time. Walking after
// parsing gives us direct access to noteIndex and processImage closures.
```

Note: The `util` import may not be needed if the compiler flags it. Remove it if so. The `parser.WithAutoHeadingID()` option generates `id` attributes on headings, matching the Ruby `with_toc_data: true` option.

```go
package images

import (
	"crypto/rand"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"os"
	"path/filepath"
	"strings"
)

// Entry is a single cached image record, matching the Ruby index.json format.
type Entry struct {
	FileName string `json:"file_name"`
	PageUID  string `json:"page_uid"`
}

// Cache manages downloaded images on disk.
type Cache struct {
	assetsPath string
}

// NewCache creates a Cache rooted at assetsPath.
func NewCache(assetsPath string) *Cache {
	return &Cache{assetsPath: assetsPath}
}

// Get returns the cached entry for a URL, downloading it if not present.
func (c *Cache) Get(imageURL, pageUID string) (Entry, error) {
	idx := c.loadIndex()
	if entry, ok := idx[imageURL]; ok {
		return entry, nil
	}

	// Download the image.
	body, originalName, err := c.download(imageURL)
	if err != nil {
		return Entry{}, fmt.Errorf("downloading %s: %w", imageURL, err)
	}

	fileName := randomName(extFrom(originalName))
	entry := Entry{FileName: fileName, PageUID: pageUID}

	// Write image file.
	dir := filepath.Join(c.assetsPath, pageUID)
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return Entry{}, err
	}
	if err := os.WriteFile(filepath.Join(dir, fileName), body, 0o644); err != nil {
		return Entry{}, err
	}

	// Update index.
	idx[imageURL] = entry
	c.saveIndex(idx)
	return entry, nil
}

// CopyTo copies a cached image to a destination directory.
func (c *Cache) CopyTo(entry Entry, destDir string) error {
	src := filepath.Join(c.assetsPath, entry.PageUID, entry.FileName)
	dst := filepath.Join(destDir, entry.FileName)
	if err := os.MkdirAll(destDir, 0o755); err != nil {
		return err
	}
	data, err := os.ReadFile(src)
	if err != nil {
		return err
	}
	return os.WriteFile(dst, data, 0o644)
}

func (c *Cache) download(imageURL string) ([]byte, string, error) {
	// Try CleanShot-style redirect first: append "+" to URL.
	if isCleanShotURL(imageURL) {
		return c.downloadCleanShot(imageURL)
	}
	return c.downloadDirect(imageURL)
}

func (c *Cache) downloadCleanShot(imageURL string) ([]byte, string, error) {
	client := &http.Client{
		CheckRedirect: func(req *http.Request, via []*http.Request) error {
			return http.ErrUseLastResponse
		},
	}
	resp, err := client.Get(imageURL + "+")
	if err != nil {
		return nil, "", err
	}
	resp.Body.Close()

	if resp.StatusCode < 300 || resp.StatusCode >= 400 {
		// Not a redirect — fall back to direct download.
		return c.downloadDirect(imageURL)
	}

	directURL := resp.Header.Get("Location")
	if directURL == "" {
		return c.downloadDirect(imageURL)
	}

	originalName := nameFromDisposition(directURL)
	body, _, err := c.downloadDirect(directURL)
	if err != nil {
		return nil, "", err
	}
	if originalName != "" {
		return body, originalName, nil
	}
	return body, nameFromURL(directURL), nil
}

func (c *Cache) downloadDirect(imageURL string) ([]byte, string, error) {
	resp, err := http.Get(imageURL)
	if err != nil {
		return nil, "", err
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		return nil, "", fmt.Errorf("HTTP %d", resp.StatusCode)
	}
	body, err := io.ReadAll(resp.Body)
	if err != nil {
		return nil, "", err
	}
	return body, nameFromURL(imageURL), nil
}

func isCleanShotURL(u string) bool {
	return strings.Contains(u, "cleanshot.com")
}

func nameFromURL(rawURL string) string {
	parsed, err := url.Parse(rawURL)
	if err != nil {
		return "image"
	}
	base := filepath.Base(parsed.Path)
	if base == "" || base == "." || base == "/" {
		return "image"
	}
	return base
}

// nameFromDisposition extracts filename from a URL's response-content-disposition query param.
func nameFromDisposition(rawURL string) string {
	parsed, err := url.Parse(rawURL)
	if err != nil {
		return ""
	}
	disp := parsed.Query().Get("response-content-disposition")
	if disp == "" {
		return ""
	}
	parts := strings.SplitN(disp, "filename=", 2)
	if len(parts) < 2 {
		return ""
	}
	return parts[1]
}

func (c *Cache) indexPath() string {
	return filepath.Join(c.assetsPath, "index.json")
}

func (c *Cache) loadIndex() map[string]Entry {
	data, err := os.ReadFile(c.indexPath())
	if err != nil {
		return make(map[string]Entry)
	}
	var idx map[string]Entry
	if err := json.Unmarshal(data, &idx); err != nil {
		return make(map[string]Entry)
	}
	return idx
}

func (c *Cache) saveIndex(idx map[string]Entry) {
	data, err := json.MarshalIndent(idx, "", "  ")
	if err != nil {
		return
	}
	os.WriteFile(c.indexPath(), data, 0o644)
}

func randomName(ext string) string {
	b := make([]byte, 16)
	rand.Read(b)
	return hex.EncodeToString(b) + ext
}

func extFrom(name string) string {
	ext := strings.ToLower(filepath.Ext(name))
	if ext == ".jpeg" {
		return ".jpg"
	}
	if ext == "" {
		return ".jpg"
	}
	return ext
}
```

```go
package build

import (
	"bytes"
	"embed"
	"fmt"
	"html/template"
	"log"
	"os"
	"path/filepath"
	"strings"
	"time"

	"github.com/dreikanter/notescli/note"
	"github.com/dreikanter/npub/internal/config"
	"github.com/dreikanter/npub/internal/images"
	"github.com/dreikanter/npub/internal/page"
	"github.com/dreikanter/npub/internal/render"
)

//go:embed all:../../templates
var templateFS embed.FS

//go:embed ../../style.css
var styleCSS []byte

// Build generates the entire static site.
func Build(cfg config.Config) error {
	// 1. Scan notes store.
	notes, err := note.Scan(cfg.NotesPath)
	if err != nil {
		return fmt.Errorf("scanning notes: %w", err)
	}

	// 2. Build note pages (public only).
	cache := images.NewCache(cfg.AssetsPath)
	var notePages []page.NotePage
	noteIndex := make(map[string]string) // ID -> public path

	for _, n := range notes {
		data, err := os.ReadFile(filepath.Join(cfg.NotesPath, n.RelPath))
		if err != nil {
			log.Printf("warning: cannot read %s: %v", n.RelPath, err)
			continue
		}

		fm := note.ParseFrontmatterFields(data)
		if !fm.Public {
			continue
		}

		slug := slugify(fm.Slug)
		if slug == "" {
			slug = slugify(fm.Title)
		}

		uid := n.Date + "_" + n.ID
		title := fm.Title
		if title == "" {
			title = uid
		}
		// Remove backticks from title (matches Ruby behavior).
		title = strings.ReplaceAll(title, "`", "")

		publishedAt, err := parseDate(n.Date)
		if err != nil {
			log.Printf("warning: cannot parse date for %s: %v", n.RelPath, err)
			continue
		}

		np := page.NotePage{
			UID:         uid,
			ShortUID:    n.ID,
			Slug:        slug,
			Title:       title,
			Description: fm.Description,
			Tags:        fm.Tags,
			PublishedAt: publishedAt,
			SiteRootURL: cfg.SiteRootURL,
		}
		notePages = append(notePages, np)
		noteIndex[n.ID] = np.PublicPath()
	}

	page.SortNotePages(notePages)

	// 3. Render Markdown for each note (second pass, now that noteIndex is complete).
	for i := range notePages {
		np := &notePages[i]
		relPath := findRelPath(notes, np.ShortUID)
		data, err := os.ReadFile(filepath.Join(cfg.NotesPath, relPath))
		if err != nil {
			continue
		}

		body := note.StripFrontmatter(data)

		processImage := func(src string) (string, error) {
			entry, err := cache.Get(src, np.UID)
			if err != nil {
				return "", err
			}
			np.Attachments = append(np.Attachments, page.Attachment{
				FileName: entry.FileName,
				PageUID:  entry.PageUID,
			})
			return entry.FileName, nil
		}

		rendered, err := render.Render(body, noteIndex, processImage)
		if err != nil {
			log.Printf("warning: render error for %s: %v", np.UID, err)
			continue
		}
		np.Body = string(rendered)
	}

	// 4. Load templates.
	tmpl, err := loadTemplates()
	if err != nil {
		return fmt.Errorf("loading templates: %w", err)
	}

	// 5. Write all pages.
	allTags := page.AllTags(notePages)

	// Write style.css.
	if err := writeFile(cfg.BuildPath, "style.css", styleCSS); err != nil {
		return err
	}

	// Write index page.
	if err := writeIndex(tmpl, cfg, notePages, allTags); err != nil {
		return err
	}

	// Write feed.
	if err := writeFeed(tmpl, cfg, notePages); err != nil {
		return err
	}

	// Write note pages, redirects, and copy attachments.
	for _, np := range notePages {
		if err := writeNotePage(tmpl, cfg, np, notePages); err != nil {
			return err
		}
		if err := writeRedirect(tmpl, cfg, np); err != nil {
			return err
		}
		// Copy image attachments.
		for _, att := range np.Attachments {
			noteDir := filepath.Join(cfg.BuildPath, np.UID, np.Slug)
			if err := cache.CopyTo(images.Entry{FileName: att.FileName, PageUID: att.PageUID}, noteDir); err != nil {
				log.Printf("warning: cannot copy attachment %s: %v", att.FileName, err)
			}
		}
	}

	// Write tag pages.
	for _, tag := range allTags {
		if err := writeTagPage(tmpl, cfg, tag, notePages, allTags); err != nil {
			return err
		}
	}

	return nil
}

func findRelPath(notes []note.Note, id string) string {
	for _, n := range notes {
		if n.ID == id {
			return n.RelPath
		}
	}
	return ""
}

func loadTemplates() (*template.Template, error) {
	return template.ParseFS(templateFS, "templates/*.html", "templates/*.xml")
}

type layoutData struct {
	Config  config.Config
	Page    pageInfo
	Content template.HTML
}

type pageInfo struct {
	Title              string
	MetaDescription    string
	CanonicalPath      string
	PublishedAt        *time.Time
	PublishedAtISO     string
	PublishedAtFormatted string
}

func newPageInfo(title, description, canonicalPath string, publishedAt *time.Time) pageInfo {
	pi := pageInfo{
		Title:           title,
		MetaDescription: description,
		CanonicalPath:   canonicalPath,
		PublishedAt:     publishedAt,
	}
	if publishedAt != nil {
		pi.PublishedAtISO = publishedAt.Format("2006-01-02")
		pi.PublishedAtFormatted = publishedAt.Format("2 January 2006")
	}
	return pi
}

func renderWithLayout(tmpl *template.Template, cfg config.Config, pi pageInfo, contentTemplate string, contentData any) ([]byte, error) {
	var contentBuf bytes.Buffer
	if err := tmpl.ExecuteTemplate(&contentBuf, contentTemplate, contentData); err != nil {
		return nil, fmt.Errorf("rendering %s: %w", contentTemplate, err)
	}

	data := layoutData{
		Config:  cfg,
		Page:    pi,
		Content: template.HTML(contentBuf.String()),
	}
	var buf bytes.Buffer
	if err := tmpl.ExecuteTemplate(&buf, "layout.html", data); err != nil {
		return nil, fmt.Errorf("rendering layout: %w", err)
	}
	return buf.Bytes(), nil
}

func writeNotePage(tmpl *template.Template, cfg config.Config, np page.NotePage, allNotes []page.NotePage) error {
	related := page.RelatedTo(allNotes, np)
	pi := newPageInfo(np.Title, np.Description, np.CanonicalPath(), &np.PublishedAt)

	noteData := struct {
		Note    page.NotePage
		Related []page.NotePage
	}{Note: np, Related: related}

	content, err := renderWithLayout(tmpl, cfg, pi, "note.html", noteData)
	if err != nil {
		return err
	}
	return writeFile(cfg.BuildPath, np.LocalPath(), content)
}

func writeRedirect(tmpl *template.Template, cfg config.Config, np page.NotePage) error {
	rp := page.RedirectPage{
		UID:        np.UID,
		RedirectTo: "/" + np.PublicPath(),
	}
	var buf bytes.Buffer
	if err := tmpl.ExecuteTemplate(&buf, "redirect.html", rp); err != nil {
		return fmt.Errorf("rendering redirect: %w", err)
	}
	return writeFile(cfg.BuildPath, rp.LocalPath(), buf.Bytes())
}

func writeIndex(tmpl *template.Template, cfg config.Config, notePages []page.NotePage, allTags []string) error {
	pi := newPageInfo("", "", "/", nil)
	indexData := struct {
		NotePages []page.NotePage
		Tags      []string
	}{NotePages: notePages, Tags: allTags}
	content, err := renderWithLayout(tmpl, cfg, pi, "index.html", indexData)
	if err != nil {
		return err
	}
	return writeFile(cfg.BuildPath, "index.html", content)
}

func writeTagPage(tmpl *template.Template, cfg config.Config, tag string, notePages []page.NotePage, allTags []string) error {
	tp := page.TagPage{Tag: tag}
	tagged := page.TaggedPages(notePages, tag)
	pi := newPageInfo(tag, "", tp.CanonicalPath(), nil)
	tagData := struct {
		CurrentTag  string
		Tags        []string
		TaggedPages []page.NotePage
	}{CurrentTag: tag, Tags: allTags, TaggedPages: tagged}
	content, err := renderWithLayout(tmpl, cfg, pi, "tag.html", tagData)
	if err != nil {
		return err
	}
	return writeFile(cfg.BuildPath, tp.LocalPath(), content)
}

func writeFeed(tmpl *template.Template, cfg config.Config, notePages []page.NotePage) error {
	feedData := struct {
		Config    config.Config
		NotePages []page.NotePage
	}{Config: cfg, NotePages: notePages}
	var buf bytes.Buffer
	if err := tmpl.ExecuteTemplate(&buf, "feed.xml", feedData); err != nil {
		return fmt.Errorf("rendering feed: %w", err)
	}
	return writeFile(cfg.BuildPath, "feed.xml", buf.Bytes())
}

func writeFile(buildPath, relPath string, data []byte) error {
	full := filepath.Join(buildPath, relPath)
	if err := os.MkdirAll(filepath.Dir(full), 0o755); err != nil {
		return err
	}
	return os.WriteFile(full, data, 0o644)
}

func parseDate(date string) (time.Time, error) {
	if len(date) < 8 {
		return time.Time{}, fmt.Errorf("date too short: %s", date)
	}
	return time.Parse("20060102", date[:8])
}

func slugify(s string) string {
	s = strings.ToLower(strings.TrimSpace(s))
	var result []byte
	for _, c := range s {
		if (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9') {
			result = append(result, byte(c))
		} else if c == ' ' || c == '-' || c == '_' {
			if len(result) > 0 && result[len(result)-1] != '-' {
				result = append(result, '-')
			}
		}
	}
	return strings.Trim(string(result), "-")
}
```

```go
func loadConfig(cmd *cobra.Command, cfgPath string) (config.Config, error) {
	if cfgPath == "" {
		cfgPath = os.Getenv("NPUB_CONFIG")
	}
	if cfgPath == "" {
		cfgPath = "npub.yml"
	}

	// Collect env overrides.
	envKeys := []string{
		"NOTES_PATH", "NPUB_ASSETS_PATH", "NPUB_BUILD_PATH",
		"NPUB_SITE_ROOT_URL", "NPUB_SITE_NAME", "NPUB_AUTHOR_NAME",
	}
	envOverrides := make(map[string]string)
	for _, key := range envKeys {
		if v := os.Getenv(key); v != "" {
			envOverrides[key] = v
		}
	}

	// Collect flag overrides (only for flags that were explicitly set).
	flagNames := []string{"notes-path", "assets-path", "out", "url", "site-name", "author"}
	flagOverrides := make(map[string]string)
	for _, name := range flagNames {
		if cmd.Flags().Changed(name) {
			v, _ := cmd.Flags().GetString(name)
			flagOverrides[name] = v
		}
	}

	return config.Load(cfgPath, envOverrides, flagOverrides)
}
```
